Mathematical summary:
For $n=100$ Gillespie paths, we compute $\mathbf{m}_N^{(l)}$, $\mathcal{G}_N^{(l)}$ and $\mathcal{V}_N^{(l)}$ , then average to get $\mathbb{E}[\mathbf{m}^{(l)}]$, $\mathbb{E}[\mathcal{G}^{(l)}]$, $\mathbb{E}[\mathcal{V}^{(l)}]$.\\

Then, $\sigma(\mathcal{M}^{(l)}(t)) = \sqrt{\int_0^t \mathbb{E}[\mathcal{V}^{(l)}(w)]dw}$, giving dispersion $\mathcal{D}^{(l)}_N = \sigma(\mathcal{M}^{(l)})/\mathbb{E}[\mathbf{m}^{(l)}]$ with derivative: $\frac{d\mathcal{D}^{(l)}_N}{dt} = \frac{\mathbb{E}[\mathbf{m}^{(l)}] \mathbb{E}[\mathcal{V}^{(l)}] - 2(\sigma(\mathcal{M}^{(l)}))^2 \mathbb{E}[\mathcal{G}^{(l)}]}{2\sigma(\mathcal{M}^{(l)}) (\mathbb{E}[\mathbf{m}^{(l)}])^2}$.

$$\frac{d\mathcal{D}^{(l)}N}{dt} = \frac{\mathbb{E}[\mathbf{m}^{(l)}(t)] \mathbb{E}[\mathcal{V}^{(l)}(t)] - 2\left(\int_0^t \mathbb{E}[\mathcal{V}^{(l)}(w)]dw\right) \mathbb{E}[\mathcal{G}^{(l)}(t)]}{2\sqrt{\int_0^t \mathbb{E}[\mathcal{V}^{(l)}(w)]dw} \cdot (\mathbb{E}[\mathbf{m}^{(l)}(t)])^2}$$

\begin{lstlisting}
ALGORITHM: Dispersion Time Derivative Analysis (Equation 5.17)
INPUT:
  - N_values = [$N_1$, $N_2$, ..., $N_k$] # Population sizes
  - n_runs = 100 # Number of stochastic runs
  - params = {$\beta$, $\gamma$, $\alpha$, $\lambda$, probabilities} # SIHRS parameters
  - T, dt # Time span and step
OUTPUT:
  - Dispersion: $\sigma(M^{(l)}) / \mathbb{E}[m^{(l)}]$ for l $\in$ {S, I, H}
  - Derivative: $\frac{d}{dt} [\sigma(M^{(l)}) / \mathbb{E}[m^{(l)}]]$
-------------------------------------------------------------
FOR each $N \in$ N_values DO:
  // Step 1: Initialize storage for all runs
  // Prepares matrices to store time-series data from simulations.
  Initialize: m_all, G_all, V_all as (n_runs $\times$ length(t)) matrices
 
  // Step 2: Run multiple stochastic simulations

  FOR run = 1 to n_runs DO:
 
    // 2a. Gillespie simulation
    // Simulates stochastic trajectories of compartment counts.
    (S(t), I(t), H(t), R(t), D(t)) $\leftarrow$ GILLESPIE(N, $\beta$, params)
   
    // 2b. Store compartment fractions
    // Normalizes counts to fractions for scale-independent analysis.
    m_S[run, :] $\leftarrow$ S(t)/N
    m_I[run, :] $\leftarrow$ I(t)/N
    m_H[run, :] $\leftarrow$ H(t)/N
   
    // 2c. Calculate drift G$^{(l)}$ using Eq. 4.2 (analytic formulas)
    G_S[run, :] $\leftarrow -\beta \cdot S \cdot I + \lambda \cdot R$
    G_I[run, :] $\leftarrow \beta \cdot S \cdot I - \gamma \cdot I$
    G_H[run, :] $\leftarrow \gamma \cdot p_{IH} \cdot I - \alpha \cdot H$
   
    // 2d. Calculate infinitesimal variance V$^{(l)}$ using Eq. 4.4
    // Quantifies stochastic fluctuations scaled by 1/N per path.
    V_S[run, :] $\leftarrow$ (1/N) $\cdot (\beta \cdot p_{SI} \cdot S \cdot I + \lambda \cdot p_{RS} \cdot R)$
    V_I[run, :] $\leftarrow$ (1/N) $\cdot (\gamma \cdot I \cdot (p_{IH} + p_{IR} + p_{ID}) + \beta \cdot p_{SI} \cdot S \cdot I)$
    V_H[run, :] $\leftarrow$ (1/N) $\cdot (\gamma \cdot p_{IH} \cdot I + \alpha \cdot H \cdot (p_{HR} + p_{HD}))$
 
  END FOR
 
  // Step 3: Average across all paths to get expected values
  // Computes ensemble averages for m, G, V.
  $\mathbb{E}[m_S]$ $\leftarrow$ mean(m_S, over runs)
  $\mathbb{E}[m_I]$ $\leftarrow$ mean(m_I, over runs)
  $\mathbb{E}[m_H]$ $\leftarrow$ mean(m_H, over runs)
 
  $\mathbb{E}[G_S]$ $\leftarrow$ mean(G_S, over runs)
  $\mathbb{E}[G_I]$ $\leftarrow$ mean(G_I, over runs)
  $\mathbb{E}[G_H]$ $\leftarrow$ mean(G_H, over runs)
 
  $\mathbb{E}[V_S]$ $\leftarrow$ mean(V_S, over runs)
  $\mathbb{E}[V_I]$ $\leftarrow$ mean(V_I, over runs)
  $\mathbb{E}[V_H]$ $\leftarrow$ mean(V_H, over runs)
 
  // Step 4: Calculate dispersion for each compartment
  // Computes relative dispersion and its derivative for S, I, H.
  FOR each compartment l $\in$ {S, I, H}:
 
    // 4a. Cumulative variance (martingale variance)
    // Integrates expected variance over time using trapezoidal rule.
    $\int \mathbb{E}[V^{(l)}]$ $\leftarrow \int_0^t \mathbb{E}[V^{(l)}](w) \, dw$ // Using trapezoidal rule
   
    // 4b. Standard deviation of martingale
    // Takes square root for martingale noise std dev.
    $\sigma(M^{(l)})$ $\leftarrow \sqrt{\int \mathbb{E}[V^{(l)}]}$
   
    // 4c. Dispersion
    // Calculates noise over mean fraction.
    Disp$^{(l)}$ $\leftarrow \sigma(M^{(l)}) / \mathbb{E}[m^{(l)}]$
   
    // 4d. Time derivative via Equation 5.17
    // Derives rate of change of dispersion.
    numerator $\leftarrow \mathbb{E}[m^{(l)}] \cdot \mathbb{E}[V^{(l)}] - 2 \cdot (\sigma(M^{(l)}))^2 \cdot \mathbb{E}[G^{(l)}]$
    denominator $\leftarrow$ 2 $\cdot \sigma(M^{(l)}) \cdot (\mathbb{E}[m^{(l)}])^2$
   
    dDisp$^{(l)}$/dt $\leftarrow$ numerator / denominator
   
    // Handle singularities (division by zero, t=0)
    // Flags invalid points with NaN.
    WHERE (denominator = 0 OR $\sigma(M^{(l)})$ = 0):
      dDisp$^{(l)}$/dt $\leftarrow$ NaN
 
  END FOR
 
  // Stores results for current N.
  Store: {Disp$^{(l)}$, dDisp$^{(l)}$/dt} for this N
END FOR
// Visualizes results across N and compartments.
PLOT: Dispersion and derivatives for all N values and compartments
-------------------------------------------------------------
SUBROUTINE: GILLESPIE(N, $\beta$, params)
  // Sets initial compartment counts.
  Initialize: S, I, H, R, D from initial conditions
  t $\leftarrow$ 0
 
  // Loops until time exceeds T.
  WHILE t < T DO:
    // Computes current transition rates.
    r_SI $\leftarrow (\beta / N) \cdot S \cdot I \cdot p_{SI}$
    r_IR $\leftarrow \gamma \cdot I \cdot p_{IR}$
    r_IH $\leftarrow \gamma \cdot I \cdot p_{IH}$
    r_ID $\leftarrow \gamma \cdot I \cdot p_{ID}$
    r_HR $\leftarrow \alpha \cdot H \cdot p_{HR}$
    r_HD $\leftarrow \alpha \cdot H \cdot p_{HD}$
    r_RS $\leftarrow \lambda \cdot R \cdot p_{RS}$
   
    r_total $\leftarrow \sum$(all rates)
   
    // Breaks if no events possible.
    IF r_total = 0 THEN BREAK
   
    // Samples waiting time and advances t.
    $\tau \leftarrow - \ln(U_1)/r_{total}$ // $U_1 \sim \mathrm{Uniform}(0,1)$
    t $\leftarrow$ t + $\tau$
   
    // Selects and executes transition.
    $U_2 \sim \mathrm{Uniform}(0, r_{total})$
    Execute transition based on $U_2$
   
    // Records current state.
    Record: (S, I, H, R, D, t)
 
  END WHILE
 
  // Interpolates to uniform time grid.
  RETURN: Interpolated trajectories on uniform grid [0, dt, 2dt, ..., T]
END SUBROUTINE
\end{lstlisting}
\textbf{Key Mathematical Points:}
\begin{enumerate}
  \item \textbf{Path-Average-First}: Calculate G, V, m for each individual stochastic path, then average $\to$ ensures proper expected values
  \item \textbf{Martingale Variance}: $\mathrm{Var}(\mathcal{M}^{(l)}(t)) = \int_0^t \mathbb{E}[\mathcal{V}^{(l)}(w)] \, dw$
  \item \textbf{Equation 5.17}:
  \[
  \frac{d}{dt}\left[\frac{\sigma(M^{(l)})}{\mathbb{E}[m^{(l)}]}\right] = \frac{\mathbb{E}[m^{(l)}] \cdot \mathbb{E}[V^{(l)}] - 2(\sigma(M^{(l)}))^2 \cdot \mathbb{E}[G^{(l)}]}{2 \cdot \sigma(M^{(l)}) \cdot (\mathbb{E}[m^{(l)}])^2}
  \]
  \item \textbf{Averaging order matters}: $\mathbb{E}[G] \neq \frac{d}{dt}\mathbb{E}[m]$ due to nonlinearity
\end{enumerate}
